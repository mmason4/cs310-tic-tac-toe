    public boolean makeMark(int row, int col) {
        if(xTurn){
            if(isValidSquare(row,col) && !isSquareMarked(row,col)) {
		grid[row][col] = Mark.X;
		xTurn = false;
		return true;
	}
        }
	else if(!xTurn){
            if(isValidSquare(row,col) && !isSquareMarked(row,col)) {

                xTurn = true;
		return true;
	}
        }	
        
        return false;
        
	}
